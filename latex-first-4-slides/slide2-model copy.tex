\begin{frame}{Model: Luminous Robots}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.52\textwidth}
Simple robots equipped with \textbf{visible light colors}.

\vspace{0.2cm}
They are \textbf{autonomous}, \textbf{anonymous}, \textbf{homogeneous}, with
\textbf{no memory}, \textbf{no compass}, \textbf{no communication},
\textbf{limited visibility}, and they operate in \textbf{synchronous LCM cycles}.

\begin{block}{LCM Phase Flow}
\begin{itemize}
  \item \textbf{Look}: capture local neighborhood.
  \item \textbf{Compute}: decide next movement from local rule.
  \item \textbf{Move}: move synchronously to next cell.
\end{itemize}
\end{block}

\begin{block}{Controls in Demo}
Play / Pause, Slower / Faster, and step-by-step phase progression.
\end{block}
