\begin{frame}{Model: Algorithm}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.52\textwidth}
\textbf{Algorithm = set of rules}

\vspace{0.2cm}
\textbf{Rule components}
\begin{itemize}
  \item View
  \item Direction
  \item Color
\end{itemize}

\textbf{Two-rule example}
\begin{block}{Rule 1}
Center=B, Right=R $\Rightarrow$ Right, B
\end{block}
\begin{block}{Rule 2}
Center=R, Left=B $\Rightarrow$ Right, R
\end{block}

\column{0.48\textwidth}
\begin{block}{LCM in grid}
\begin{itemize}
  \item Look: observe
  \item Compute: match rule
  \item Move: execute
\end{itemize}
\end{block}
