\begin{itemize}
  \item \textbf{1. The Problem: Grid Exploration}
        \newline {\footnotesize\textcolor{softgray}{[Bramas et al. 2021, Sardar et al. 2024]}}
        \newline $\rightarrow$ \textit{Limits:} Exploration without obstacle (luminous) / Exploration with obstacle (other models)
  \vspace{0.4cm}

  \item \textbf{2. Generation: Hand vs. Synthesis}
        \newline {\footnotesize\textcolor{softgray}{[Bonnet et al. 2014, Millet et al. 2014]}}
        \newline $\rightarrow$ \textit{Limits:} Error-prone / Combinatorial explosion
  \vspace{0.4cm}

  \item \textbf{3. Validation: Formal Verification}
        \newline {\footnotesize\textcolor{softgray}{[Bérard et al. 2016, Courtieu et al. 2022]}}
        \newline $\rightarrow$ \textit{Limits:} Manual effort / Huge execution space
\end{itemize}
