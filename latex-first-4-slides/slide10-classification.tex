\begin{frame}{\ModuleFrameTitle{3}{Phase 4: Automated Classification (Concept)}}

\textbf{Goal:} Group generated algorithms into behavioral equivalence classes.

\vspace{0.4cm}
\begin{itemize}\setlength\itemsep{0.4cm}
    \item \textbf{C1: Rules} \textcolor{softgray}{\scriptsize (Syntactic comparison of rule sets)}
    \item \textbf{C2: Rounds} \textcolor{softgray}{\scriptsize (Time to complete one full exploration cycle)}
    \item \textbf{C3: Energy} \textcolor{softgray}{\scriptsize (Total movements + color changes per cycle)}
    \item \textbf{C4: Path} \textcolor{softgray}{\scriptsize (Physical trajectory on the grid)}
\end{itemize}

\end{frame}
