\begin{frame}{\ModuleFrameTitle{3}{Phase 1: Scenario-Guided Synthesis}}

\textbf{The Bottleneck:} Brute-force generation explodes (e.g., ${\sim}10^{147}$ combinations).

\vspace{0.4cm}

\uncover<2->{
\textbf{Our Solution: Bounded Goals}
\vspace{0.1cm}
\begin{itemize}\setlength\itemsep{2pt}
    \item Define \textbf{Starting ($S$)} and \textbf{Ending ($E$)} local configurations
    \item Restrict search space via \textbf{Bounding Box} and max \textbf{$k$ rounds}
\end{itemize}
}

\vspace{0.4cm}
